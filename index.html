function countPositivesSumNegatives(input) {
if (input === null || input.length === 0) {
return [];
}
let amount = 0;
let sum = 0;
for (let i = 0; i < input.length; i++) {
if (input[i] > 0) {
amount++;
} else if (input[i] < 0) {
sum += input[i];
}
}
return [amount, sum];
}

function findNb(m) {
let n = 0;
let sum = 0;
while (sum < m) {
n++;
sum += Math.pow(n, 3);
}
return sum === m ? n : -1;
}
